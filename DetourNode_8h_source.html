118  inline dtNode* pop()
119  {
120  dtNode* result = m_heap[0];
121  m_size--;

123  return result;
124  }

126  inline void push(dtNode* node)
127  {
128  m_size++;
129  bubbleUp(m_size-1, node);
130  }
131 
132  inline void modify(dtNode* node)
133  {
134  for (int i = 0; i < m_size; ++i)
135  {
136  if (m_heap[i] == node)
137  {

139  return;
140  }
141  }
142  }
